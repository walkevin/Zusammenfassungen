\subsection{Kovarianz und Korrelation}
\begin{tabular}{p{4cm} >{$}p{16cm}<{$}}
Kovarianz		& \Cov(X,Y) = \E(XY) - \E(X)\E(Y)\\
			& \Cov(X,X) = \V(X)\\
Korrelation		& \Corr(X,Y) := \rho_{XY} := \frac{\Cov(X,Y)}{\sigma_X \sigma_Y}\quad \in[-1,1]\\
Rechenregeln		& \Cov(a + bX, c + dY) = bd\Cov(X,Y)\\
			& \Corr(a + bX, c + dY) = \sgn(b)\sgn(d) \Corr(X,Y)\\
			& \quad\V(X + Y) = \V(X) + \V(Y) + 2 \Cov(X,Y)\\
			& \quad\E(XY) = \E(X)\E(Y) + \Cov(X,Y)\\
			& X,Y \text{ unabhängig } \Rightarrow \Corr(X,Y) = 0 \Rightarrow \Cov(X,Y) = 0\\
			& \Corr(X,Y) = +1 \Leftrightarrow Y = a + bX,\quad a \in \mathbb{R}, b \boldsymbol{>} 0\\
			& \Corr(X,Y) = -1 \Leftrightarrow Y = a + bX,\quad a \in \mathbb{R}, b \boldsymbol{<} 0\\
Lineare Prognose	& \hat{Y} = \mu_Y + \frac{\Cov(X,Y)}{\V(X)}(X-\mu_X)\\
Prognosefehler		& \E \big( (Y-\hat{Y})^2 \big) = (1-\rho_{XY}^2) \V(Y)
\end{tabular}
