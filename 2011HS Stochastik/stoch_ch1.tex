\section{Begriff der Wahrscheinlichkeit}
\begin{tabular}{p{4cm} >{$}p{16cm}<{$}}
Rechenregeln	& \begin{array}[t]{l}
            	  \P (A) \geq 0\\
		  \P (\Omega) = 1\\
		  \P ( A \cup B ) = \P (A) + \P (B) - \P(A \cap B) \quad \text{vgl. Inklusion - Exklusion}\\
		  \P (B \setminus A ) = \P(B) - \P(A)\\
            	  \end{array}\\
Unabhängigkeit von Ereignissen	& \begin{array}[t]{l}
                              	  \text{Zwei Ereignisse $A$ und $B$ sind stochastisch unabhängig gdw.}\\
				  \P ( A \cap B) = \P (A) \P (B)\\
				  \P ( A | B ) = \P (A) \Leftrightarrow \P(B | A) = \P(B)\\
                              	  \end{array}\\
\end{tabular}
