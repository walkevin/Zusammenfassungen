Theorem 1.3	& \text{Every LP is either infeasible, unbounded or it has an optimal solution}\\
The Dual	& \text{Primal: } \begin{array}[t]{|ll|}
				    \text{max}	& c^Tx\\
				    \text{s.t.}	& Ax  \leq b\\
						& x \geq \mathbf{0} 
				  \end{array}
		  \text{  Dual: } \begin{array}[t]{|ll|}
				    \text{min}	& b^Ty\\
				    \text{s.t.}	& A^Ty  \geq c\\
						& y \geq \mathbf{0} 
				  \end{array}\\
Weak Duality	& c^Tx \leq b^Ty \text{ for any pair of primal and dual feasible solutions $x,y$}\\
Strong Duality	& \text{$c^T\bar{x} = b^T\bar{y} \Leftrightarrow $ LP has {\bfseries optimal }solution $\bar{x}$ and its dual has opt.sol as well, $\bar{y}$}\\
Complementary Slackness Conditions	& \begin{array}[t]{ll}
                                  	   \multicolumn{2}{l}{\text{The following conditions are equivalent for opt.sol $\bar{x}$ and $\bar{y}$}}\\
					   (a)  &\text{$\bar{x}$ and $\bar{y}$ are optimal solutions}\\
					   (b)  &c^T\bar{x} = b^T\bar{y}\\
					   (c)  &\bar{y}^T(b-A\bar{x}) = 0 \wedge \bar{x}^T(A^T\bar{y} -c) = 0\\
					   (c') &\bar{y_i}(b-A\bar{x})_i = 0 \forall i \wedge \bar{x_j}^T(A^T\bar{y} -c)_j = 0 \forall j\\
					   (c'')& \bar{y}_i > 0 \Rightarrow (b-A\bar{x})_i = 0 \forall i \wedge \bar{x}_j > 0 \Rightarrow (A^T\bar{y} -c)_j = 0\forall j
                                  	  \end{array}\\
Farkas' Lemma	& \text{A System of inequalities }Ax \leq b, x \geq \mathbf{0} \text{ is {\bfseries infeasible}}\Leftrightarrow \exists y \geq \mathbf{0}: A^Ty \geq \mathbf{0} \wedge b^Ty < 0\\
Theorem 2.5	& \text{A LP max }c^Tx, Ax\leq b, x\geq \mathbf{0} \text{ is {\bfseries unbounded}} \Leftrightarrow \exists \text{sol. $x$ and } z \geq \mathbf{0}: Az \leq \mathbf{0} \wedge c^Tz > 0\\
Remark II (29.7.11)	& \begin{array}[t]{l}
                	   \text{If dual is unbounded $\Rightarrow$ primal is infeasible and vice versa, because}\\
			   \text{Weak Duality: }c^Tx \leq b^Ty;\enspace y \rightarrow -\infty
                	  \end{array}\\
Equivalences		& A^Tx = b \Leftrightarrow A^Tx \leq b \wedge -A^Tx \leq -b\\
			& x_j \in \mathbb{R} \Leftrightarrow x_j = x_j' - x_j'' \wedge x_j', x_j'' \geq 0\\
Example			& \text{Primal: }\begin{array}{|ll|}
			   \text{max}	& c^Tx\\
			   \text{s.t.}	& Ax = b\\
					& x \geq \mathbf{0}
			  \end{array} =
			  \begin{array}{|ll|}
			   \text{max}	& c^Tx\\
			   \text{s.t.}	& Ax \leq b\\
					&-Ax \leq -b\\
					& x \geq \mathbf{0}
			  \end{array}\\
			& \Rightarrow\text{Dual: }\begin{array}{|ll|}
			   \text{min}	& b^Ty' - b^Ty''\\
			   \text{s.t.}	& A^Ty' - A^Ty'' \leq c\\
					& y',y'' \geq \mathbf{0}
			  \end{array} =
			  \begin{array}{|ll|}
			   \text{min}	& b^T (y' - y'') = b^Ty\\
			   \text{s.t.}	& A^T (y' - y'') = A^Ty \leq c\\
					& y \in \mathbb{R}
			  \end{array}\\
